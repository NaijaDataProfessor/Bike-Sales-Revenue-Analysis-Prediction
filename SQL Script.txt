SELECT * FROM store_sale;

--- Total Transactions
SELECT COUNT(*) AS total_transaction
FROM (
		SELECT DISTINCT *
		FROM store_sale
        ) AS distinct_transactions;
        
--- Total Revenue
SELECT CAST(SUM(revenue) AS DECIMAL(10,2)) AS total_revenue
FROM (
		SELECT DISTINCT *
		FROM store_sale
        ) AS distinct_transactions;
	
--- Staging the raw_data into a dataset
CREATE TEMPORARY TABLE dataset LIKE store_sale;

INSERT INTO dataset
SELECT DISTINCT *
FROM store_sale;

--- Top Countries by Revenue 
SELECT country, 
CAST(SUM(revenue) AS DECIMAL(10,2)) AS total_revenue,
CAST(SUM(profit) AS DECIMAL(10,2)) AS total_profit
FROM dataset
GROUP BY country
ORDER BY total_revenue DESC
LIMIT 10;

--- Top States by Revenue 
SELECT state, 
CAST(SUM(revenue) AS DECIMAL(10,2)) AS total_revenue,
CAST(SUM(profit) AS DECIMAL(10,2)) AS total_profit
FROM dataset
GROUP BY state
ORDER BY total_revenue DESC
LIMIT 10;

--- Top Product Category by Revenue 
SELECT category, 
CAST(SUM(revenue) AS DECIMAL(10,2)) AS total_revenue
FROM dataset
GROUP BY category
ORDER BY total_revenue DESC
LIMIT 10;


--- Top Sub-Category by Revenue 
SELECT sub_category, 
CAST(SUM(revenue) AS DECIMAL(10,2)) AS total_revenue
FROM dataset
GROUP BY sub_category
ORDER BY total_revenue DESC
LIMIT 10;

--- Most Profitable Products 
SELECT product,
CAST(SUM(profit) AS  DECIMAL(10,2)) AS total_profit
FROM dataset
GROUP BY product
ORDER BY  total_profit DESC
LIMIT 10;


--- Revenue Distribution across age groups 
SELECT age_group,
CAST(SUM(revenue) AS  DECIMAL(10,2)) AS total_profit
FROM dataset
GROUP BY age_group
ORDER BY total_profit DESC;


--- Revenue Distribution across gender
SELECT CASE
			WHEN gender = "M" THEN "Male"
            WHEN gender = "F" THEN "Female"
		END AS genders,
CAST(SUM(revenue) AS  DECIMAL(10,2)) AS total_profit
FROM dataset
GROUP BY genders
ORDER BY total_profit DESC;


--- Average Monthly Revenue Trend 
SELECT MONTH(STR_TO_DATE(puchase_date, '%m/%d/%Y'))  AS   month_no, 
MONTHNAME(STR_TO_DATE(puchase_date, '%m/%d/%Y'))  AS   month_name,
CAST(AVG(revenue) AS DECIMAL(10,2)) AS average_revenue 
FROM dataset
GROUP BY month_no, month_name
ORDER BY month_no; 


--- Yearly Revenue Trend 
SELECT  YEAR(STR_TO_DATE(puchase_date, '%m/%d/%Y'))  AS   purchase_year,
CAST(SUM(revenue) AS DECIMAL(10,2)) AS total_revenue 
FROM dataset
GROUP BY purchase_year
ORDER BY purchase_year;
